35def _clamav_items() -> List[ClamavItem]: 

36 item_list = [] 

37 for section in config.sections(): 

38 if not config.is_enabled(section): 

39 continue 

40 for option in config.options(section): 

41 check = config.integrity_check(section) 

42 max_age = config.max_age(section) 

43 max_size = config.max_size(section) 

44 if option.startswith('url_'): 

45 url = config.get(section, option) 

46 path: str = urlparse(url).path 

47 slash_pos = path.rfind('/') # returns -1 if not found 

48 path = os.path.join(config.local_dir(section), path[slash_pos + 1:]) 

49 item = ClamavItem(section, option, url, check, path, max_age, max_size) 

50 item_list.append(item) 

51 return item_list

54class ClamavRefresh: 

55 def __init__(self, args) -> None: 

56 self.args = args 

57 

58 def refresh(self, ci: ClamavItem) -> bool: 

59 """Refresh a single ClamAV item. 

60 

61 :param ci: Item to refresh. 

62 :return: True if new payload data was written, False otherwise. 

63 """ 

64 try: 

65 if self.args.force: 

66 log.debug(f'{ci.url} refresh forced') 

67 elif not RefreshLog.is_outdated(ci.url, ci.max_age): 

68 log.debug(f'{ci.url} below max age') 

69 return False 

70 status, digest = get_digest(ci) 

71 if not status: 

72 return False 

73 if RefreshLog.digest_matches(ci.url, digest): 

74 log.debug(f'{ci.url} unchanged') 

75 RefreshLog.update(ci.url, digest) # Update timestamp 

76 return False 

77 status, payload = get_payload(ci) 

78 if not status: 

79 return False 

80 status, message = check_integrity(payload, ci.check, digest) 

81 if not status: 

82 log.warning(f'{ci.url} {message}') 

83 return False 

84 log.info(f'Updating {ci.path}') 

85 with open(ci.path, 'wb') as f: 

86 f.write(payload) 

87 RefreshLog.update(ci.url, digest) # Update digest and timestamp 

88 except OSError as e: # pragma: no cover 

89 log.exception(e) 

90 return True

92 def refresh_all(self) -> int: 

93 count = 0 

94 for ci in _clamav_items(): 

95 if self.refresh(ci): 

96 count += 1 

97 _exec = config.on_update_exec() 

98 if count > 0 and _exec: 

99 try: 

100 timeout = config.on_update_timeout() 

101 p: CompletedProcess = run(_exec, capture_output=True, encoding='utf-8', shell=True, timeout=timeout) 

102 if p.stdout: 

103 log.info(p.stdout) 

104 if p.stderr: 

105 log.error(p.stderr) 

106 except CalledProcessError as e: # pragma: no cover 

107 log.exception(e) 

108 return count
